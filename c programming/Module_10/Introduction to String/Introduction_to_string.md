#### M. Replace MinMax
```
#include <stdio.h>
#include <string.h>
int main()
{

    int n;
    scanf("%d", &n);
    int a[n];

    for (int i = 0; i < n; i++)
    {
        scanf("%d", &a[i]);
    }

    int min = a[0];

    int max = a[n - 1];

    for (int i = 0; i < n; i++)
    {
        if (min > a[i])
        {
            min = a[i];
        }

        if (max < a[i])
        {
            max = a[i];
        }
    }
    
    int ind_min;
    int ind_max;

    for (int i = 0; i < n; i++)
    {
        if (a[i] == min)
        {
            ind_min = i;
            break;
        }
         
    }

    for (int i = 0; i < n; i++)
    {
        if (a[i] == max)
        {
            ind_max = i;
            break;
        }
        
    }
    
    //int ct = ind_min;
    a[ind_min] = max;
    a[ind_max] = min;

    for (int  i = 0; i < n; i++)
    {
        printf("%d ", a[i]);
    }
    

    return 0;
}
```

#### G. Conversion 
```
#include <stdio.h>
#include <string.h>
int main()
{
    char a[100001];
    fgets(a, 100001, stdin);

    char space = ' ';

    for (int i = 0; i < strlen(a); i++)
    {
        if (a[i] == ',')
        {
            a[i] = space;
        }
        else if (a[i] >= 'a' && a[i] <= 'z')
        {
            a[i] = a[i] - 32;
        }
        else if (a[i] >= 'A' && a[i] <= 'Z')
        {
            a[i] = a[i] + 32;
        }
    }

    printf("%s", a);

    return 0;
}
```

#### I. Palindrome
```
#include <stdio.h>
#include <string.h>
int main()
{
    char a[1005];
    scanf("%s", &a);
    int ans = 1;
    int i = 0, j = strlen(a) - 1;
    while (i < j)
    {
        int tmp = a[i];
        a[i] = a[j];
        a[j] = tmp;
        if (a[i] != a[j])
        {
            ans = 0;
            break;
        }
        i++;
        j--;
    }

    if (ans == 1)
    {
        printf("YES");
    }
    else if (ans == 0)
    {
        printf("NO");
    }

    return 0;
}
```
